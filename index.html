const tg = window.Telegram.WebApp; tg.expand(); const grid = document.getElementById('grid'); const footer = document.getElementById('footer'); const totalSum = document.getElementById('total-sum'); let cart = {}; // Твои товары СТРОГО по списку из GitHub const products = [ { id: 1, name: 'Орешки со сгущенкой', price: 1.10, min: 10, img: 'oreshki.png' }, { id: 2, name: 'Орешки с орехом', price: 1.50, min: 10, img: 'oreshki sguchenka oreh.png' }, { id: 3, name: 'Трубочка', price: 2.20, min: 5, img: 'trubochka1.jpg' }, { id: 4, name: 'Картошка', price: 1.80, min: 5, img: 'kartoshka.jpg' }, { id: 5, name: 'Шок. колбаса', price: 2.00, min: 1, img: 'shokoladnaya kolbasa.jpg' } ]; function render() { if (!grid) return; grid.innerHTML = ''; products.forEach(p => { const count = cart[p.id] || 0; const card = document.createElement('div'); card.className = 'product-card'; card.innerHTML = `
${p.name}
Заказ от ${p.min} шт.
${p.price.toFixed(2)} BYN
${count}
`; grid.appendChild(card); }); } window.change = (id, delta) => { const p = products.find(x => x.id === id); let count = cart[id] || 0; if (delta === 1) { count = (count === 0) ? p.min : count + 1; } else { if (count === p.min) count = 0; else if (count > p.min) count -= 1; } if (count <= 0) delete cart[id]; else cart[id] = count; render(); updateFooter(); }; function updateFooter() { let sum = 0; Object.keys(cart).forEach(id => { const p = products.find(x => x.id == id); sum += p.price * cart[id]; }); if (sum > 0) { footer.style.display = 'block'; totalSum.innerText = sum.toFixed(2) + ' BYN'; } else { footer.style.display = 'none'; } } window.clearCart = () => { cart = {}; render(); updateFooter(); }; window.sendOrder = () => { if (Object.keys(cart).length === 0) return; tg.sendData(JSON.stringify(cart)); tg.close(); }; render();
